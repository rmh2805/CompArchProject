Micro Execute for each ALU Instructions.
Assume OP1-OP4 values are already set. 

uADD:
   ur1 <- OP2 + OP3
   goto writeback

uSUB:
   ur1 <- OP2 - OP3
   goto writeback

uAND:
   ur1 <- OP2 & OP3
   goto writeback

uOR:
   ur1 <- OP2 | OP3
   goto writeback 

uXOR:
   ur1 <- OP2 ^ OP3
   goto writeback 

uINC:
   goto uADD

uDEC:
   goto uSUB

uCMP:
   ur1 <- ~OP2
   goto writeback

uSLT:
   uSUB(OP2, OP3)
   ur1 < 0
      ur1 <- 1
   ur1 = 0
