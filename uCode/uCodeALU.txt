Assume OP1-OP4 values are already set. 

ADD:
   uR0 <- OP2Val + OP3Val, goto Writeback

SUB:
   uR0 <- OP2Val - OP3Val, goto Writeback

AND:
   uR0 <- OP2Val & OP3Val, goto Writeback

OR:
   uR0 <- OP2Val | OP3Val, goto Writeback

XOR:
   uR0 <- OP2Val ^ OP3Val, goto Writeback

INC:
   uR0 <- OP1Val + 1, goto Writeback

DEC:
   uR0 <- OP1Val - 1, goto Writeback
